{{- $title := "" }} {{- /* Vox: for sections with numeric names, use the raw slug to avoid Hugo humanizing "2026" into "2026s". .File is nil for auto-generated sections so we check the section path segment instead. */ -}} {{- if and .IsSection (findRE `^\d+$` (path.Base .RelPermalink)) -}} {{- $title = path.Base .RelPermalink -}} {{- else if and .IsSection (findRE `^[wW]\d+$` (path.Base .RelPermalink)) -}} {{- $title = path.Base .RelPermalink | upper -}} {{- else if .LinkTitle }} {{- $title = .LinkTitle }} {{- else if .Title }} {{- $title = .Title }} {{- else if and .IsSection .File }} {{- $title = path.Base .File.Dir | humanize | title }} {{- else if and .IsPage .File }} {{- $title = .File.BaseFileName | humanize | title }} {{- end }} {{ return $title -}}
